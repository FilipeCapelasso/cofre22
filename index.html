function doPost(e) {
  try {
    var sheet = SpreadsheetApp
      .openById("1s3OO7_TR8gIBOfrFuprTM0uCQMlpZDTIvZtXDhJbSPQ")
      .getSheets()[0];
    var d = JSON.parse(e.postData.contents);

    // restante calculado no servidor
    var restante = d.salario -
      (d.investimento + d.reserva + d.lazer + d.carro);

    sheet.appendRow([
      d.data,          // A - data
      d.salario,       // B - salário
      d.investimento,  // C - investimento
      d.reserva,       // D - reserva
      d.lazer,         // E - lazer
      d.carro,         // F - carro
      restante         // G - restante
    ]);

    return ContentService
      .createTextOutput(JSON.stringify({"status":"success"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({
        "status":"error",
        "message":err.toString()
      })).setMimeType(ContentService.MimeType.JSON);
  }
}
