function doPost(e) {
  try {
    var sheet = SpreadsheetApp
      .openById("1s3OO7_TR8gIBOfrFuprTM0uCQMlpZDTIvZtXDhJbSPQ")
      .getSheets()[0];

    // Lê o corpo JSON enviado pelo site
    var raw  = e.postData.contents;
    var d    = JSON.parse(raw);

    // Garante que os campos existem
    var dataReg  = d["data"]         || "";
    var salario  = Number(d["salario"])  || 0;
    var invest   = Number(d["investimento"]) || 0;
    var reserva  = Number(d["reserva"])  || 0;
    var lazer    = Number(d["lazer"])    || 0;
    var carro    = Number(d["carro"])    || 0;
    var restante = salario - (invest + reserva + lazer + carro);

    // Insere linha nas colunas A → G
    sheet.appendRow([
      dataReg,   // A — data
      salario,   // B — salário
      invest,    // C — investimento
      reserva,   // D — reserva
      lazer,     // E — lazer
      carro,     // F — carro
      restante   // G — restante
    ]);

    return ContentService
      .createTextOutput('{"status":"success"}')
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService
      .createTextOutput(
        '{"status":"error","msg":"' + err + '"}')
      .setMimeType(ContentService.MimeType.JSON);
  }
}
